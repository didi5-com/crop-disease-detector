📦 2. Auto Data Loader
utils/data_loader.py
from tensorflow.keras.preprocessing.image import ImageDataGenerator
import config

def get_data():
    train_gen = ImageDataGenerator(
        rescale=1./255,
        rotation_range=20,
        zoom_range=0.2,
        horizontal_flip=True
    )

    val_gen = ImageDataGenerator(rescale=1./255)

    train_data = train_gen.flow_from_directory(
        config.TRAIN_DIR,
        target_size=(config.IMG_SIZE, config.IMG_SIZE),
        batch_size=config.BATCH_SIZE,
        class_mode='categorical'
    )

    val_data = val_gen.flow_from_directory(
        config.VAL_DIR,
        target_size=(config.IMG_SIZE, config.IMG_SIZE),
        batch_size=config.BATCH_SIZE,
        class_mode='categorical'
    )

    return train_data, val_data
🤖 3. Smarter Model Builder (Transfer Learning + Fine-Tune)
utils/model_builder.py
from tensorflow.keras.applications import MobileNetV2
from tensorflow.keras.layers import Dense, GlobalAveragePooling2D
from tensorflow.keras.models import Model

def build_model(num_classes):
    base_model = MobileNetV2(weights='imagenet', include_top=False, input_shape=(224,224,3))

    # Freeze most layers (fast + stable)
    for layer in base_model.layers[:-20]:
        layer.trainable = False

    x = base_model.output
    x = GlobalAveragePooling2D()(x)
    x = Dense(128, activation='relu')(x)
    output = Dense(num_classes, activation='softmax')(x)

    model = Model(inputs=base_model.input, outputs=output)
    return model
🧠 4. Automated Training Pipeline
train.py
from utils.data_loader import get_data
from utils.model_builder import build_model
from tensorflow.keras.callbacks import ModelCheckpoint, EarlyStopping
import config

train_data, val_data = get_data()

model = build_model(train_data.num_classes)

model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])

checkpoint = ModelCheckpoint(
    config.MODEL_PATH,
    monitor='val_accuracy',
    save_best_only=True,
    verbose=1
)

early_stop = EarlyStopping(
    monitor='val_loss',
    patience=3,
    restore_best_weights=True
)

model.fit(
    train_data,
    validation_data=val_data,
    epochs=config.EPOCHS,
    callbacks=[checkpoint, early_stop]
)

🔍 5. Smart Inference Engine
inference.py
import numpy as np
import cv2
from tensorflow.keras.models import load_model
import config
from utils.disease_info import get_solution

model = load_model(config.MODEL_PATH)

def preprocess(img_path):
    img = cv2.imread(img_path)
    img = cv2.resize(img, (224,224))
    img = img / 255.0
    return np.reshape(img, (1,224,224,3))

def predict(img_path, class_indices):
    img = preprocess(img_path)
    pred = model.predict(img)

    idx = np.argmax(pred)
    confidence = float(np.max(pred))

    labels = list(class_indices.keys())
    disease = labels[idx]

    solution = get_solution(disease)

    return disease, confidence, solution

def get_solution(disease):
    solutions = {
        "Healthy": "No action needed",
        "Leaf Blight": "Apply copper-based fungicide",
        "Rust": "Use sulfur spray",
        "Mosaic Virus": "Remove infected plants"
    }
    return solutions.get(disease, "Consult agricultural expert")
🌐 7. Flask App (Now Smarter)
app.py
from flask import Flask, render_template, request
import os
from inference import predict
from utils.data_loader import get_data

app = Flask(__name__)
UPLOAD_FOLDER = 'static/uploads'
app.config['UPLOAD_FOLDER'] = UPLOAD_FOLDER

# Load class labels automatically
train_data, _ = get_data()
class_indices = {v:k for k,v in train_data.class_indices.items()}

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/predict', methods=['POST'])
def predict_route():
    file = request.files['file']
    filepath = os.path.join(app.config['UPLOAD_FOLDER'], file.filename)
    file.save(filepath)

    disease, confidence, solution = predict(filepath, class_indices)

    return render_template('result.html',
                           disease=disease,
                           confidence=round(confidence*100,2),
                           solution=solution,
                           image=filepath)

if __name__ == '__main__':
    app.run(debug=True)
