🌐 3. API Client (Core Integration)
utils/api_client.py
import requests

API_KEY = "YOUR_API_KEY_HERE"

def get_disease_info(image_path):
    url = "https://api.kindwise.com/v1/identification"

    headers = {
        "Api-Key": API_KEY
    }

    files = {
        "images": open(image_path, "rb")
    }

    data = {
        "modifiers": ["crops_fast", "similar_images"],
        "plant_language": "en",
        "disease_details": ["description", "treatment", "classification"]
    }

    try:
        response = requests.post(url, headers=headers, files=files, data=data)
        result = response.json()

        if "result" not in result:
            return None

        disease = result["result"]["disease"]["suggestions"][0]

        return {
            "name": disease.get("name", "Unknown"),
            "probability": disease.get("probability", 0),
            "description": disease.get("details", {}).get("description", ""),
            "treatment": disease.get("details", {}).get("treatment", {}).get("chemical", ""),
            "images": disease.get("similar_images", [])
        }

    except Exception as e:
        print("API ERROR:", e)
        return None
🔍 4. Upgrade Your Inference Engine
Replace inference.py with:
import numpy as np
import cv2
from tensorflow.keras.models import load_model
import config
from utils.disease_info import get_solution
from utils.api_client import get_disease_info

model = load_model(config.MODEL_PATH)

def preprocess(img_path):
    img = cv2.imread(img_path)
    img = cv2.resize(img, (224,224))
    img = img / 255.0
    return np.reshape(img, (1,224,224,3))

def predict(img_path, class_indices):
    img = preprocess(img_path)
    pred = model.predict(img)

    idx = np.argmax(pred)
    confidence = float(np.max(pred))

    labels = list(class_indices.keys())
    local_disease = labels[idx]

    # 🔥 CALL API
    api_data = get_disease_info(img_path)

    if api_data:
        return {
            "disease": api_data["name"],
            "confidence": api_data["probability"] * 100,
            "description": api_data["description"],
            "treatment": api_data["treatment"],
            "images": api_data["images"]
        }
    else:
        # fallback to local model
        return {
            "disease": local_disease,
            "confidence": confidence * 100,
            "description": "No API data available",
            "treatment": get_solution(local_disease),
            "images": []
        }
